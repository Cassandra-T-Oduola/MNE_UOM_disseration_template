\documentclass[12pt]{MNE_UOM_thesis}

%load any additional packages
\usepackage{amssymb}
\usepackage[utf8]{inputenc}
\usepackage{graphicx}
\usepackage[acronym]{glossaries}
\usepackage{subfiles}
\usepackage{ragged2e}

%NOTE: This template has an implementation of a subfiles environment, hence each tex can be recompiled on its own for faster recompilation time.

%input macros (i.e. write your own macros file called mymacros.tex

\title{Codimension-Two Free Boundary Problems}  %Title of your thesis
\author{John Borg}                              %your name
\department{Department name}                    %your department
\degree{Doctor of Philosophy}                   %the degree
\degreedate{Trinity 1998}                       %the degree date

%paths to the same images folder relative to the root and the "sections" folders.
\graphicspath{{images/}{../images/}} 

%generate list of glossary
%Note: It is important to clear cache file in share latex to regenerate new glossary
\makeglossaries 

%end the preamble and start the document
\begin{document}

%this baselineskip gives sufficient line spacing for an examiner to easily
%markup the thesis with comments
\baselineskip=18pt plus1pt

%set the number of sectioning levels that get number and appear in the contents
\setcounter{secnumdepth}{3}
\setcounter{tocdepth}{3}

\maketitle                                  % create a title page from the preamble info
\begin{romanpages}                          % start roman page numbering
\include{./frontmatter/dedication}          % include a dedication.tex file
\include{./frontmatter/copyright}           % include a copyright.tex file
\include{./frontmatter/declaration}         % include a declariation.tex file
\include{./frontmatter/abstract}            % include the abstract
\include{./frontmatter/acknowledgements}    % include an acknowledgements.tex file
\include{./frontmatter/glossary}             % include an glossary.tex file

\tableofcontents                            % generate and include a table of contents
\listoffigures                              % generate and include a list of figures
\addcontentsline{toc}{section}{List of Figures}
\listoftables                               % generate and include a list of tables
\addcontentsline{toc}{section}{List of Tables}
\printglossary
\printglossary[title=\begin{center} \Large \bfseries LIST OF ACRONYMS \end{center}, type=\acronymtype]
\addcontentsline{toc}{section}{List of Acronyms}

\end{romanpages}                            % end roman page numbering

%%%%%%%%%%%%%%%%%%%%%%  END of Front Matter  %%%%%%%%%%%%%%%%%%%%%%

%now include the files of latex for each of the chapters etc
\chapter{First chapter}

    \section{Introduction}
    \subfile{chapters/introduction}         %path to tex file

    \section{Second section}
    \subfile{chapters/section2}             %path to tex file
    
    \section{Conclusion}
    \subfile{chapters/conclusions}          %path to tex file

%%%%%%%%%%%%%%%%%%%%%%  START of Back Matter  %%%%%%%%%%%%%%%%%%%%%%

%now enable appendix numbering format and include any appendices
\appendix

%next line adds the Bibliography to the contents page
\addcontentsline{toc}{chapter}{Bibliography}

%uncomment next line to change bibliography name to references
%\renewcommand{\bibname}{References}
